## Постройте пайплайн для переливки данных в ClickHouse с возможностью регулярного или одноразового выполнения.

Созданный dag:
```py
from datetime import datetime, timedelta, date
# import pandas as pd
import requests
from airflow import DAG
from airflow.decorators import task
from airflow_clickhouse_plugin.hooks.clickhouse import ClickHouseHook
import logging

BASE_URL = 'https://iss.moex.com/iss/statistics/engines/stock/markets/bonds/bondization.json'
PAGE_LIMIT = 100
CON_CH = 'ch_yc'

default_args = {
    'owner': 'tmyu',
    'depends_on_past': False,
    'retries': 1,
    'retry_delay': timedelta(minutes=2),
}

with DAG(
    dag_id='moex_coupons_to_clickhouse',
    default_args=default_args,
    description='Загрузка купонов облигаций из MOEX в ClickHouse с пагинацией',
    schedule_interval='0 8 * * *',
    start_date=datetime(2026, 6, 1),
    catchup=False,
    tags=['moex', 'clickhouse', 'bonds'],
) as dag:

    @task
    def fetch_all_coupons(from_date: str = '2026-05-01', till_date: str = '2026-06-30'):
        """Постраничное получение всех купонов из MOEX и объединение в pandas DataFrame"""
        start = 0
        all_data = []
        columns = None
        
        while True:
            params = {
                'from': from_date,
                'till': till_date,
                'start': start,
                'limit': PAGE_LIMIT,
                'iss.only': 'coupons',
                'iss.meta': 'off'
            }
            
            try:
                response = requests.get(BASE_URL, params=params, timeout=30)
                response.raise_for_status()
                data = response.json()
                
                # Извлекаем секцию coupons
                coupons_section = data.get('coupons')
                if not coupons_section:
                    logging.warning("Секция 'coupons' не найдена в ответе")
                    break
                
                # Получаем колонки (при первом запросе)
                if columns is None:
                    columns = coupons_section.get('columns')
                    if not columns:
                        raise ValueError("Не найдены колонки в ответе MOEX")
                
                # Получаем данные текущей страницы
                batch = coupons_section.get('data', [])
                if not batch:
                    break
                
                all_data.extend(batch)
                logging.info(f"Загружено записей: {len(all_data)} (start={start})")
                    
                start += len(batch)
                
            except requests.exceptions.RequestException as e:
                logging.error(f"Ошибка при запросе: {e}")
                raise
        
        if not all_data:
            raise ValueError("Не получено ни одной записи купонов")
        
        # Передаём данные для следующего таска
        return all_data, columns

    @task
    def create_table_if_not_exists():
        """Создаёт таблицу в ClickHouse на основе структуры данных MOEX"""
        hook = ClickHouseHook(clickhouse_conn_id=CON_CH)
        
        create_query = """
        CREATE TABLE IF NOT EXISTS moex_bond_coupons (
            isin Nullable(String),
            name Nullable(String),
            issuevalue Nullable(Float64),
            coupondate Date,
            recorddate Nullable(Date),
            startdate Nullable(Date),
            initialfacevalue Nullable(Float64),
            facevalue Nullable(Float64),
            faceunit Nullable(String),
            value Nullable(Float64),
            valueprc Nullable(Float64),
            value_rub Nullable(Float64),
            secid String,
            primary_boardid Nullable(String),
            loaded_at DateTime DEFAULT now()
        ) ENGINE = MergeTree()
        ORDER BY (secid, coupondate)
        """
        hook.execute(create_query)
        logging.info("Таблица moex_bond_coupons проверена/создана")

    @task
    def insert_into_clickhouse(coupons):
        """Вставляет данные в ClickHouse"""
        hook = ClickHouseHook(clickhouse_conn_id=CON_CH)

        all_data, columns = coupons
        
        # Список колонок в том порядке, как они определены в таблице
        columns_order = [
            'isin', 'name', 'issuevalue', 'coupondate', 'recorddate', 
            'startdate', 'initialfacevalue', 'facevalue', 'faceunit', 
            'value', 'valueprc', 'value_rub', 'secid', 'primary_boardid'
        ]

        existing_cols = [col for col in columns_order if col in columns]
        if not existing_cols:
            error_msg = "Отсутствуют колонки для вставки в данных"
            logging.error(error_msg)
            raise ValueError(error_msg)

        indexes = [columns.index(col) for col in existing_cols]
        
        # Преобразуем в список кортежей
        data_tuples = [tuple(
            row[idx] if not existing_cols[i].endswith('date') or row[idx] is None
            else date.fromisoformat(row[idx]
            ) for i, idx in enumerate(indexes)) for row in all_data]
        
        # Вставка пачками по 10000 строк
        chunk_size = 10000
        total_inserted = 0
        
        for i in range(0, len(data_tuples), chunk_size):
            chunk = data_tuples[i:i+chunk_size]
            hook.execute(
                sql=f"""
                    INSERT INTO moex_bond_coupons 
                    ({', '.join(existing_cols)}) 
                    VALUES
                """,
                params=chunk
            )
            total_inserted += len(chunk)
            logging.info(f"Вставлено строк: {total_inserted} из {len(data_tuples)}")
        
        return f"Успешно вставлено {total_inserted} записей"

    # Определяем flow зависимостей
    coupons = fetch_all_coupons()
    create_table = create_table_if_not_exists()
    insert_task = insert_into_clickhouse(coupons)
    insert_task.set_upstream(create_table)
```
